## Logistic regression

For this regression we choose to fit a quasibinomial distribution

```r
mod <- glm(formula = group ~ ., family =  quasibinomial(link = "logit"), data = training_data %>% clean_names())
mod
```

## ROC
And the resulting ROC curve give us :

```r
prob_cmml <- function(mod, NEWX, RDWSDfL, PDWfL) {
  if(any(is.na(c(NEWX, RDWSDfL, PDWfL)))) { return(NA)} else {
    return(1/(1+exp(-(mod$coefficients["rdw_sd_f_l"]*as.numeric(RDWSDfL)
                      +mod$coefficients["ne_wx"]*as.numeric(NEWX)
                      +mod$coefficients["pdw_f_l"]*as.numeric(PDWfL)
                      +mod$coefficients[1]))))
  }
}

res <- training_data %>% rowwise() %>% mutate(score = prob_cmml(mod, `[NE-WX]`, `RDW-SD(fL)`, `PDW(fL)`))

rre <- plot.roc(res$group, res$score, levels = as.factor(c(0,1)), direction = '<', percent = T, ci = T, of = "thresholds", thresholds = "best", print.thres = "best")
```
